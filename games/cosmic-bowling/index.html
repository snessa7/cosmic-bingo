const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); // Game state let pins = []; let ball = null; let currentFrame = 1; let currentRoll = 1; let totalScore = 0; let frameScores = []; let isRolling = false; let power = 0; let powerDirection = 1; let powerInterval = null; // Pin class class Pin { constructor(x, y, index) { this.x = x; this.y = y; this.index = index; this.radius = 15; this.standing = true; this.falling = false; this.fallAngle = 0; this.glowIntensity = 0; } draw() { ctx.save(); ctx.translate(this.x, this.y); if (this.falling) { ctx.rotate(this.fallAngle); this.fallAngle += 0.1; if (this.fallAngle > Math.PI / 2) { this.standing = false; this.falling = false; } } if (this.standing || this.falling) { // Pin body ctx.fillStyle = '#ffffff'; ctx.beginPath(); ctx.arc(0, 0, this.radius, 0, Math.PI * 2); ctx.fill(); // Glow effect this.glowIntensity = Math.sin(Date.now() * 0.003 + this.index) * 0.5 + 0.5; ctx.shadowBlur = 20 * this.glowIntensity; ctx.shadowColor = '#ffff00'; ctx.fill(); ctx.shadowBlur = 0; // Number ctx.fillStyle = '#000'; ctx.font = 'bold 12px Arial'; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.fillText(this.index + 1, 0, 0); } ctx.restore(); } checkCollision(ball) { if (!this.standing || this.falling) return false; const dx = ball.x - this.x; const dy = ball.y - this.y; const distance = Math.sqrt(dx * dx + dy * dy); if (distance < this.radius + ball.radius) { this.falling = true; // Add some physics to the ball ball.vx *= 0.7; ball.vy *= 0.7; return true; } return false; } } // Ball class class Ball { constructor(x, y) { this.x = x; this.y = y; this.vx = 0; this.vy = 0; this.radius = 12; this.trail = []; this.active = false; } update() { if (!this.active) return; this.x += this.vx; this.y += this.vy; // Friction this.vx *= 0.99; this.vy *= 0.985; // Wall collision if (this.x - this.radius < 0 || this.x + this.radius > canvas.width) { this.vx = -this.vx * 0.8; this.x = Math.max(this.radius, Math.min(canvas.width - this.radius, this.x)); } // Add to trail this.trail.push({ x: this.x, y: this.y }); if (this.trail.length > 20) { this.trail.shift(); } // Stop if too slow if (Math.abs(this.vx) < 0.1 && Math.abs(this.vy) < 0.5) { this.active = false; endRoll(); } } draw() { // Draw trail ctx.strokeStyle = 'rgba(255, 255, 0, 0.3)'; ctx.lineWidth = this.radius * 2; ctx.lineCap = 'round'; ctx.beginPath(); this.trail.forEach((point, index) => { if (index === 0) { ctx.moveTo(point.x, point.y); } else { ctx.lineTo(point.x, point.y); } }); ctx.stroke(); // Draw ball const gradient = ctx.createRadialGradient(this.x, this.y, 0, this.x, this.y, this.radius); gradient.addColorStop(0, '#ffff88'); gradient.addColorStop(0.5, '#ffaa00'); gradient.addColorStop(1, '#ff6600'); ctx.fillStyle = gradient; ctx.beginPath(); ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2); ctx.fill(); // Glow ctx.shadowBlur = 30; ctx.shadowColor = '#ffaa00'; ctx.fill(); ctx.shadowBlur = 0; } } // Initialize pins function setupPins() { pins = []; const startX = canvas.width / 2; const startY = 150; const spacing = 35; // Pin layout (standard 10-pin) const layout = [ [0, 0], // Pin 1 [-0.5, 1], [0.5, 1], // Pins 2, 3 [-1, 2], [0, 2], [1, 2], // Pins 4, 5, 6 [-1.5, 3], [-0.5, 3], [0.5, 3], [1.5, 3] // Pins 7, 8, 9, 10 ]; layout.forEach((pos, index) => { const x = startX + pos[0] * spacing; const y = startY + pos[1] * spacing; pins.push(new Pin(x, y, index)); }); } // Initialize frame display function setupFrames() { const frameContainer = document.getElementById('frameScore'); frameContainer.innerHTML = ''; for (let i = 0; i < 10; i++) { const frame = document.createElement('div'); frame.className = 'frame'; frame.id = `frame${i + 1}`; frame.innerHTML = `
Frame ${i + 1}

-
`; frameContainer.appendChild(frame); } frameScores = Array(10).fill(null).map(() => ({ rolls: [], score: 0 })); } // Start power meter function startPowerMeter() { power = 0; powerDirection = 1; powerInterval = setInterval(() => { power += powerDirection * 5; if (power >= 100 || power <= 0) { powerDirection = -powerDirection; } document.getElementById('powerFill').style.width = power + '%'; }, 50); } // Stop power meter function stopPowerMeter() { if (powerInterval) { clearInterval(powerInterval); powerInterval = null; } } // Bowl the ball function bowl() { if (isRolling) return; const position = parseInt(document.getElementById('positionSlider').value); const angle = parseInt(document.getElementById('angleSlider').value); stopPowerMeter(); ball = new Ball(canvas.width / 2 + position, canvas.height - 50); ball.vx = Math.sin(angle * Math.PI / 180) * (power / 10); ball.vy = -15 - (power / 10); ball.active = true; isRolling = true; document.getElementById('bowlBtn').disabled = true; } // End roll function endRoll() { isRolling = false; // Count knocked pins const knockedPins = pins.filter(pin => !pin.standing).length; const standingPins = pins.filter(pin => pin.standing).length; // Show knocked pins if (knockedPins > 0) { const display = document.getElementById('pinsDisplay'); display.textContent = knockedPins === 10 ? 'STRIKE!' : (currentRoll === 2 && standingPins === 0) ? 'SPARE!' : `${knockedPins} pins!`; display.style.display = 'block'; setTimeout(() => { display.style.display = 'none'; }, 2000); } // Update frame score const frameIndex = currentFrame - 1; frameScores[frameIndex].rolls.push(knockedPins); // Update display updateFrameDisplay(); // Check if frame is complete if (currentRoll === 1 && knockedPins === 10) { // Strike - frame complete nextFrame(); } else if (currentRoll === 2) { // Second roll - frame complete nextFrame(); } else { // First roll, not a strike currentRoll = 2; document.getElementById('currentRoll').textContent = currentRoll; // Remove knocked pins pins = pins.filter(pin => pin.standing); // Reset ball position setTimeout(() => { ball = null; document.getElementById('bowlBtn').disabled = false; startPowerMeter(); }, 1000); } } // Move to next frame function nextFrame() { currentFrame++; currentRoll = 1; if (currentFrame > 10) { // Game over endGame(); } else { document.getElementById('currentFrame').textContent = currentFrame; document.getElementById('currentRoll').textContent = currentRoll; // Reset pins setupPins(); // Update active frame highlight document.querySelectorAll('.frame').forEach(f => f.classList.remove('active')); document.getElementById(`frame${currentFrame}`).classList.add('active'); // Reset for next roll setTimeout(() => { ball = null; document.getElementById('bowlBtn').disabled = false; startPowerMeter(); }, 1500); } } // Update frame display function updateFrameDisplay() { const frameIndex = currentFrame - 1; const frame = document.getElementById(`frame${currentFrame}`); const rolls = frameScores[frameIndex].rolls; let rollsText = rolls.map((r, i) => { if (i === 0 && r === 10) return 'X'; if (i === 1 && rolls[0] + r === 10) return '/'; return r.toString(); }).join(' '); frame.querySelector('.rolls').textContent = rollsText; // Calculate total score calculateTotalScore(); } // Calculate total score function calculateTotalScore() { let total = 0; for (let i = 0; i < frameScores.length; i++) { const frame = frameScores[i]; if (frame.rolls.length === 0) break; let frameScore = frame.rolls.reduce((a, b) => a + b, 0); // Strike bonus if (frame.rolls[0] === 10 && i < 9) { const nextFrame = frameScores[i + 1]; if (nextFrame.rolls.length > 0) { frameScore += nextFrame.rolls[0]; if (nextFrame.rolls.length > 1) { frameScore += nextFrame.rolls[1]; } else if (nextFrame.rolls[0] === 10 && i < 8) { const nextNextFrame = frameScores[i + 2]; if (nextNextFrame.rolls.length > 0) { frameScore += nextNextFrame.rolls[0]; } } } } // Spare bonus else if (frame.rolls.length === 2 && frame.rolls[0] + frame.rolls[1] === 10 && i < 9) { const nextFrame = frameScores[i + 1]; if (nextFrame.rolls.length > 0) { frameScore += nextFrame.rolls[0]; } } total += frameScore; frame.score = total; document.querySelector(`#frame${i + 1} .score`).textContent = total; } totalScore = total; document.getElementById('totalScore').textContent = total; } // End game function endGame() { const display = document.getElementById('pinsDisplay'); display.innerHTML = `

Game Over!

Final Score: ${totalScore}

`; display.style.display = 'block'; } // Reset game function resetGame() { currentFrame = 1; currentRoll = 1; totalScore = 0; document.getElementById('currentFrame').textContent = currentFrame; document.getElementById('currentRoll').textContent = currentRoll; document.getElementById('totalScore').textContent = totalScore; document.getElementById('pinsDisplay').style.display = 'none'; setupFrames(); setupPins(); ball = null; document.getElementById('bowlBtn').disabled = false; startPowerMeter(); document.getElementById(`frame1`).classList.add('active'); } // Game loop function gameLoop() { // Clear canvas ctx.fillStyle = 'rgba(0, 0, 0, 0.1)'; ctx.fillRect(0, 0, canvas.width, canvas.height); // Draw lane ctx.strokeStyle = 'rgba(255, 255, 255, 0.2)'; ctx.lineWidth = 2; ctx.beginPath(); ctx.moveTo(50, 0); ctx.lineTo(50, canvas.height); ctx.moveTo(canvas.width - 50, 0); ctx.lineTo(canvas.width - 50, canvas.height); ctx.stroke(); // Draw pins pins.forEach(pin => pin.draw()); // Update and draw ball if (ball) { ball.update(); ball.draw(); // Check collisions pins.forEach(pin => pin.checkCollision(ball)); } requestAnimationFrame(gameLoop); } // Event listeners document.getElementById('bowlBtn').addEventListener('click', bowl); // Create starry background function createStars() { const starsContainer = document.querySelector('.stars-bg'); for (let i = 0; i < 100; i++) { const star = document.createElement('div'); star.className = 'star'; star.style.left = Math.random() * 100 + '%'; star.style.top = Math.random() * 100 + '%'; star.style.animationDelay = Math.random() * 3 + 's'; starsContainer.appendChild(star); } } // Initialize game createStars(); setupFrames(); setupPins(); startPowerMeter(); gameLoop(); document.getElementById(`frame1`).classList.add('active');
